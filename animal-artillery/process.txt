Browser loads index.html

    ↓

CONSTRUCTOR → CREATE → UPDATE (repeats forever)

main.js
// This runs FIRST and tells Phaser:
const config = {
    width: window.innerWidth,    // Use full window
    height: window.innerHeight,  // Use full window  
    scale: Phaser.Scale.RESIZE,  // Auto-resize magic
    scene: [Boot, Game]          // Scenes to load (in order)
};


boot.js
class Boot extends Phaser.Scene {
    create() {
        // Immediately switches to Game scene
        this.scene.start('Game');
    }
}


game.js
